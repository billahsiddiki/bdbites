-- ============================================

-- 1. SOURCE REGISTRY
CREATE TABLE sources (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name TEXT NOT NULL,
  url TEXT NOT NULL,
  feed_type TEXT NOT NULL,          -- 'rss', 'apify', 'google_alerts'
  language TEXT DEFAULT 'bn',        -- 'bn', 'en'
  tier INT NOT NULL DEFAULT 2,       -- 1=official, 2=major, 3=secondary
  category TEXT,                     -- 'politics','business','entertainment'
  is_active BOOLEAN DEFAULT true,
  last_scraped_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT now()
);

-- 2. RAW SCRAPED ITEMS
CREATE TABLE raw_items (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  source_id UUID REFERENCES sources(id),
  source_url TEXT NOT NULL,
  url_hash TEXT UNIQUE NOT NULL,     -- SHA-256 for dedup
  title TEXT,
  body TEXT,
  image_url TEXT,
  published_at TIMESTAMPTZ,
  scraped_at TIMESTAMPTZ DEFAULT now(),
  language TEXT DEFAULT 'bn',
  processed BOOLEAN DEFAULT false
);
CREATE INDEX idx_raw_hash ON raw_items(url_hash);
CREATE INDEX idx_raw_processed ON raw_items(processed);

-- 3. CONTENT QUEUE (scored items)
CREATE TABLE content_queue (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  raw_item_id UUID REFERENCES raw_items(id),
  title TEXT NOT NULL,
  body TEXT,
  source_url TEXT NOT NULL,
  source_name TEXT,
  image_url TEXT,
  score INTEGER DEFAULT 0,
  score_emotion INTEGER,
  score_relevance INTEGER,
  score_timeliness INTEGER,
  score_shareability INTEGER,
  score_visual INTEGER,
  score_uniqueness INTEGER,
  score_safety INTEGER,
  channels TEXT[],                   -- ['the_mandate','dheu']
  language TEXT,
  format TEXT,                       -- 'news_card','quote_card', etc.
  urgency TEXT,                      -- 'immediate','next_slot','scheduled'
  hook_bangla TEXT,
  hook_english TEXT,
  deeper_layer TEXT,
  emotional_trigger TEXT,
  status TEXT DEFAULT 'pending',     -- pending,writing,designed,qa_pass,
                                     -- qa_fail,awaiting_approval,approved,
                                     -- published,rejected,error
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now()
);
CREATE INDEX idx_queue_status ON content_queue(status);
CREATE INDEX idx_queue_score ON content_queue(score DESC);

-- 4. GENERATED CONTENT (per channel × platform)
CREATE TABLE generated_content (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  queue_id UUID REFERENCES content_queue(id),
  channel TEXT NOT NULL,
  platform TEXT NOT NULL,            -- 'facebook','instagram','tiktok','youtube','x','linkedin'
  headline TEXT,
  caption TEXT,
  hashtags TEXT[],
  cta TEXT,
  image_url TEXT,
  template_id TEXT,
  bannerbear_uid TEXT,
  qa_verdict TEXT,                   -- 'pass','fail','rewrite'
  qa_notes TEXT,
  rewrite_count INT DEFAULT 0,
  approved_by TEXT,                  -- 'auto' or 'human:telegram_user'
  approved_at TIMESTAMPTZ,
  published_at TIMESTAMPTZ,
  platform_post_id TEXT,
  status TEXT DEFAULT 'draft',
  created_at TIMESTAMPTZ DEFAULT now()
);
CREATE INDEX idx_content_status ON generated_content(channel, status);

-- 5. PUBLISHED (engagement tracking)
CREATE TABLE published (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  content_id UUID REFERENCES generated_content(id),
  channel TEXT,
  platform TEXT,
  post_id TEXT,
  published_at TIMESTAMPTZ DEFAULT now(),
  likes INT DEFAULT 0,
  comments INT DEFAULT 0,
  shares INT DEFAULT 0,
  saves INT DEFAULT 0,
  reach INT DEFAULT 0,
  impressions INT DEFAULT 0,
  engagement_rate FLOAT DEFAULT 0,
  fetched_at TIMESTAMPTZ
);

-- 6. SCORING CONFIG (adjustable weights)
CREATE TABLE scoring_config (
  id SERIAL PRIMARY KEY,
  criterion TEXT UNIQUE,
  weight FLOAT,
  updated_at TIMESTAMPTZ DEFAULT now()
);
INSERT INTO scoring_config (criterion, weight) VALUES

-- 7. PROMPT VERSIONS (A/B testing)
CREATE TABLE prompt_versions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  prompt_name TEXT NOT NULL,
  prompt_type TEXT NOT NULL,         -- 'scorer','caption','qa','optimizer'
  channel TEXT,
  system_prompt TEXT NOT NULL,
  user_prompt_template TEXT NOT NULL,
  is_active BOOLEAN DEFAULT false,
  win_rate FLOAT DEFAULT 0,
  total_tests INT DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT now()
);

-- 8. PUBLISH SCHEDULE
CREATE TABLE publish_schedule (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  channel TEXT NOT NULL,
  platform TEXT NOT NULL,
  day_of_week INT,                   -- 0=Mon, 6=Sun; NULL=every day
  post_time TIME NOT NULL,
  is_active BOOLEAN DEFAULT true
);

-- 9. ERROR LOG
CREATE TABLE pipeline_errors (
  id SERIAL PRIMARY KEY,
  created_at TIMESTAMPTZ DEFAULT now(),
  workflow TEXT,
  node_name TEXT,
  error_message TEXT,
  severity TEXT NOT NULL,            -- 'CRITICAL','WARNING','INFO'
  payload JSONB,
  resolved BOOLEAN DEFAULT false
);

-- 10. OPTIMIZATION LOG
CREATE TABLE optimization_logs (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  cycle_date TIMESTAMPTZ DEFAULT now(),
  channel TEXT,
  posts_analyzed INT,
  avg_engagement FLOAT,
  top_hook_type TEXT,
  top_posting_time TEXT,
  weight_changes JSONB,
  recommendations TEXT,
  applied BOOLEAN DEFAULT false

20. How Photos Actually Work

Every news article has a hidden HTML tag: <meta property="og:image">. This is the photo that appears when you share a link on Facebook. When Apify scrapes an article, it extracts this image URL alongside the headline and text. That is your photo — a real photo of a real event, chosen by the original newsroom.
